create table browse_recipient as (select oda, parentsector_code, purposecode,
 year, 
recipientcode,fao_sector, 
sum(value) as value,
max(unitcode) as unitcode
 from usd_aggregated_table
 group by

 oda, parentsector_code, purposecode, year, recipientcode,fao_sector
 order by
 oda, parentsector_code, purposecode, year, recipientcode,fao_sector);
